! Propósito: activar posible defecto latente en compilador de Triángulo original

let
  func F () : Integer ~ 1! la función devuelve siempre el entero 1
 
  ;
  proc P (x : Integer) ~ ! P espera paso por constante
    putint (x * 2)
  end

  P (F()) ! F es una función y su resultado es entero; sirve como parámetro real constante (valor entero)
