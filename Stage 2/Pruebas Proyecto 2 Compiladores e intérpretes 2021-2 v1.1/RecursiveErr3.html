! Nombre de función repetido


let

  recursive

    func f (): Integer ~ 1

    |

    func g (b : Integer): Integer ~ 3

    |

    func f (a : Integer): Integer ~ 2 ! es inválido repetir nombres en una declaración mutuamente recursiva

  end

in

  putint(f(10)) ! aquí­ también podrí­a haber alguna dificultad

end
