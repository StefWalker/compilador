! local se elabora de manera que las declaraciones secuenciales se trabajan apropiadamente
let

  local
      var a   := 1;
      const b ~ 1 

  in
      proc P() ~ putint(a) end; ! a visible aquí
      func m(): Integer ~ b + 2 ! b visible aquí

  end

in

  P() ; ! P visible aquí
  puteol () ;
  putint (m()) ! m visible aquí

end
